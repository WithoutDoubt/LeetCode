                  window[c]++;
                  if (window[c] == need[c]) //c的数目够了，就可以了
                      valid++;

              }

  class Solution {

      bool checkInclusion(string s1, string s2) {
          unordered_map<char,int> needs, window;
          int left = 0;
          int right = 0;
          int valid = 0;
          for(char c : s1) needs[c]++;
  
          while(right < s2.size()){
              char c = s2[right];
              right++;
              if (needs.count(c)){  // c是不是属于needs中的
                  window[c]++;
                  if (needs[c] == window[c])
                      valid++;
              }

                  if(right - left == s1.size()){
                      return true;

                  left++;

                          valid--;
                  }
              }
          }

  class Solution {

          vector<int> res;
  
          unordered_map<char,int> needs,windows;

                  }
              }
          }

  class Solution {
  public:
      int lengthOfLongestSubstring(string s) {
          unordered_map<char,int> window;
          int n = s.size();
          int left = 0;
          int right = 0;
          int res = 0;
  
          while(right < n){
              char c1 = s[right];
              window[c1]++;
              right++;
  
              while(window[c1] > 1 ){ 
                  window[s[left++]]--;
                  
              }
              res = max(right-left,res);
          }
         return res;
      }
  };
